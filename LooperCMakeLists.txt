cmake_minimum_required(VERSION 3.20)
set(PLUGIN_NAME "Looper")

project(${PLUGIN_NAME} VERSION 1.0.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(JUCE_MODULES
    juce_audio_basics
    juce_audio_devices
    juce_audio_formats
    juce_audio_plugin_client
    juce_audio_processors
    juce_core
    juce_data_structures
    juce_events
    juce_graphics
    juce_gui_basics
    juce_gui_extra
)

juce_add_plugin(${PLUGIN_NAME}
    PLUGIN_NAME ${PLUGIN_NAME}
    PRODUCT_NAME ${PLUGIN_NAME}
    PLUGIN_CODE Lopr
    COMPANY_NAME "MyPlugins"
    COMPANY_WEBSITE "https://example.com"
    IS_SYNTH FALSE
    NEEDS_MIDI_INPUT TRUE
    NEEDS_MIDI_OUTPUT FALSE
    IS_MIDI_EFFECT FALSE
    EDITOR_WANTS_KEYBOARD_FOCUS FALSE
    COPY_PLUGIN_AFTER_BUILD FALSE
    PLUGIN_FORMATS Standalone VST3 AU
)

target_compile_definitions(${PLUGIN_NAME}
    PUBLIC
    JUCE_WEB_BROWSER=0
    JUCE_USE_CURL=0
    JUCE_APPLICATION_URL_STRING=""
    JUCE_APPLICATION_NAME_STRING="${PLUGIN_NAME}"
)

juce_generate_juce_header(${PLUGIN_NAME})

target_sources(${PLUGIN_NAME} PRIVATE
    LooperProcessor.cpp
    LooperProcessor.h
    LooperLayer.h
    primitives/dsp/CaptureBuffer.h
    primitives/dsp/LoopBuffer.h
    primitives/dsp/Playhead.h
    primitives/dsp/TempoInference.h
    primitives/dsp/Quantizer.h
    primitives/ui/Canvas.h
    primitives/ui/CanvasStyle.h
)

target_link_libraries(${PLUGIN_NAME} PRIVATE
    ${JUCE_MODULES}
)

set_target_properties(${PLUGIN_NAME} PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/${PLUGIN_NAME}_artefacts/Release/Standalone"
    LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/${PLUGIN_NAME}_artefacts/Release/VST3"
)
